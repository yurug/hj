\begin{frame}
\frametitle{Spécification non fonctionnelle}

\begin{itemize}
\item \textbf{Interactivité}
\only<2->{\\ $\Rightarrow$ Un modèle de calcul réactif}
\item \textbf{Sécurité}
\only<3->{\\ $\Rightarrow$ Un système d'exécution en bac à sable}
\item \textbf{Généricité vis-à-vis de l'environnement d'exécution}
\only<4->{\\ $\Rightarrow$ Un langage de description de correcteurs}
\item \textbf{Interopérabilité}
\only<5->{\\ $\Rightarrow$ Une représentation textuelle de l'état du système}
\item \textbf{Ergonomie}
\only<6->{\\ $\Rightarrow$ Une interface UNIX et une interface WEB}
\only<7->{\\ $\Rightarrow$ Un déploiement possible sur sa machine personnelle}
\item \textbf{Robustesse}
\only<8->{\\ $\Rightarrow$ Un système de journalisation automatique}
\only<9->{\\ $\Rightarrow$ Un système qui passe à l'échelle}
\only<10->{\\ $\Rightarrow$ Une plateforme de programmation sûre \verb!;-)!}
\end{itemize}

\end{frame}

\begin{frame}
\frametitle{Déroulement d'une évaluation}

\begin{enumerate}
\item L'étudiant soumet une réponse à une question.
\item Une machine virtuelle ``fraîche'' est allouée.
\item La réponse est copiée sur la machine virtuelle.
\item Les sources d'évaluation sont copiées sur la machine virtuelle.
\item Le correcteur est exécuté sur la machine.
\item Durant l'exécution, on capture la sortie standard du correcteur,
  on observe les fichiers créés, le temps d'exécution, ...
\item Une évaluation quantitative absolue est synthétisée à partir de 
  la trace d'exécution du correcteur.
\item L'évaluation relative est calculée en fonction des autres soumissions.
\item L'étudiant est notifié.
\item L'enseignant est notifié et peut apposer une évaluation qualitative. 
\end{enumerate}

\end{frame}
